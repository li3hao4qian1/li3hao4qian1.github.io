#include<bits/stdc++.h>
using namespace std;
//Optimize优化，略
int a,b,n,x[1005][1005],mx[1005][1005],mn[1005][1005],
        mxx[1005][1005],mnn[1005][1005],ans=INT_MAX;
struct s{
    int v,id;
};
inline int read() {
    //快读，略
}
//单调队列核心
inline void loadmx(int rn,int lmt,int supersecretnumber=1){
    deque<s> qd;
    for(int i=1;i<=rn;i++){
        int xx=x[supersecretnumber][i];
        while(!qd.empty()&&qd.front().id+lmt<=i){
            qd.pop_front();
        }
        while(!qd.empty()&&qd.back().v<=xx){
            qd.pop_back();
        }
        qd.push_back({xx,i});
        if(i>=lmt){
            mx[supersecretnumber][i]=qd.front().v;
        }
    }
}
inline void loadmn(int rn,int lmt,int supersecretnumber=1){
    //略
}
inline void loadmxagain(int rn,int lmt,int supersecretnumber=1){
    deque<s> qd;
    for(int i=1;i<=rn;i++){
        int xx;
        xx=mx[i][supersecretnumber];
        //略
        if(i>=lmt){
            mxx[i][supersecretnumber]=qd.front().v;
        }
    }
}
inline void loadmnagain(int rn,int lmt,int supersecretnumber=1){
    //略
}
int main(){
    a=read();
    b=read();
    n=read();
    for(int i=1;i<=a;i++){
        for(int j=1;j<=b;j++){
            x[i][j]=read();
        }
        loadmx(b,n,i);
        loadmn(b,n,i);
    }
    for(int j=1;j<=b;j++){
        loadmxagain(a,n,j);
        loadmnagain(a,n,j);
    }
    for(int i=n;i<=a;i++){
        for(int j=n;j<=b;j++){
            ans=min(ans,mxx[i][j]-mnn[i][j]);
        }
    }
    cout<<ans;
    return 0;
}
//O(n^2)

DFS可以拿20pts，这里略过不写。
明眼人你不是人都会看出这是一个DP。
我们（包括你）可以想到， 可以设dp[i][j]是考虑a[i]，且LIS长度为j时的方案数。
思索…… a[i][j]为-1时，既然不考虑，只需要继承上一个便可（即dp[i-1][j-1]）;
a[i][j]为0时，除了考虑上述情况，还要考虑接不上（也就是v[i]小于v[i-1]）的情况，
这时，可以从a[1]到a[j]的任何一个状态继承；
a[i][j]大于0时，如果他不大于j，可以从接不上的角度继承；
这时，我们还要注意一个特例，a[i]==0时，也可以从上一个继承。
AC代码
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define P 998244353
int n,a[5005],dp[5005][5005],ans;
signed main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    dp[0][0]=1;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=i;j++){
            if(a[i]==0){
                dp[i][j]=dp[i-1][j-1]+dp[i-1][j]*j;
            }else if(a[i]>0){
                if(a[i]==j){
                    dp[i][j]=dp[i-1][j-1]+dp[i-1][j];
                }if(a[i]<j){
                    dp[i][j]=dp[i-1][j];
                }
            }else{
                dp[i][j]=dp[i-1][j-1];
            }
            dp[i][j]%=P;
        }
    }
    for(int i=1;i<=n;i++){
        ans+=dp[n][i];
    }
    cout<<ans%P;
    return 0;
}

比较小的40分，可以通过bitset优化，复杂度为O(m^2*n/64)

链的分，可以通过排序，贪心选取右端点小的。

满分的贪心，考虑一个性质：两条路径的相交，一定是某一条路径的lca在另一条路径上。

选择一条路径之后，它的lca对应的子树中的其他点，是不能在通过lca走向其他的点。

优先选取lca深度最大的点。这样影响的路径最少。

选取一条路径之后，把它lca内的所有节点都标记掉，表示不可以在使用。一个点如果已经被标记了，就不用在继续往下标记了。这样每个点，只会被标记一次。
